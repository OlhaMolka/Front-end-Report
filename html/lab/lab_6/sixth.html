        const medicines = [
        { name: "Noshpa", price: 170 },
        { name: "Analgin", price: 55 },
        { name: "Quanil", price: 310 },
        { name: "Alphacholine", price: 390 },
      ];
      
      function applyDiscount(medicines) {
        const discountedMedicines = medicines.map((medicine, index) => {
          if (medicine.price > 300) {
            const discountedPrice = medicine.price * 0.7; // Знижка на 30%
            return { id: index + 1, name: medicine.name, price: discountedPrice };
          } else {
            return { id: index + 1, name: medicine.name, price: medicine.price };
          }
        });
      
        return discountedMedicines;
      }
      
      const discountedMedicines = applyDiscount(medicines);
      console.log(discountedMedicines);
